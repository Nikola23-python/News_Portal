from django.contrib.auth.models import User
from news.models import Author, Category, Post, Comment

user1 = User.objects.create_user('user1')
user2 = User.objects.create_user('user2')

author1 = Author.objects.create(user=user1)
author2 = Author.objects.create(user=user2)

category1 = Category.objects.create(name='Политика')
category2 = Category.objects.create(name='Спорт')
category3 = Category.objects.create(name='Образование')
category4 = Category.objects.create(name='Технологии')

post1 = Post.objects.create(
    author=author1,
    post_type=Post.ARTICLE,
    title='Первая статья',
    content='Содержание первой статьи...' * 20,
    rating=0
)
post1.categories.add(category1, category2)

post2 = Post.objects.create(
    author=author1,
    post_type=Post.ARTICLE,
    title='Вторая статья',
    content='Содержание второй статьи...' * 20,
    rating=0
)
post2.categories.add(category3)

news1 = Post.objects.create(
    author=author2,
    post_type=Post.NEWS,
    title='Первая новость',
    content='Содержание новости...' * 20,
    rating=0
)
news1.categories.add(category1, category4)

comment1 = Comment.objects.create(
    post=post1,
    user=user1,
    text='Комментарий 1 к первой статье',
    rating=0
)
comment2 = Comment.objects.create(
    post=post1,
    user=user2,
    text='Комментарий 2 к первой статье',
    rating=0
)
comment3 = Comment.objects.create(
    post=post2,
    user=user1,
    text='Комментарий к второй статье',
    rating=0
)
comment4 = Comment.objects.create(
    post=news1,
    user=user2,
    text='Комментарий к новости',
    rating=0
)

post1.like()  
post1.like() 
post1.dislike()

post2.like() 
post2.like()  

news1.dislike() 

comment1.like()  
comment1.like()  
comment2.dislike()  
comment3.like()  
comment4.dislike()  

author1.update_rating()
author2.update_rating()

best_author = Author.objects.order_by('-rating').first()
print(f"Лучший автор: {best_author.user.username}, рейтинг: {best_author.rating}")

best_post = Post.objects.filter(post_type=Post.ARTICLE).order_by('-rating').first()
print(f"Лучшая статья:")
print(f"Дата: {best_post.created_at}")
print(f"Автор: {best_post.author.user.username}")
print(f"Рейтинг: {best_post.rating}")
print(f"Заголовок: {best_post.title}")
print(f"Превью: {best_post.preview()}")

print("\nКомментарии к лучшей статье:")
comments = Comment.objects.filter(post=best_post).order_by('-created_at')
for comment in comments:
    print(f"{comment.created_at} | {comment.user.username} | {comment.rating} | {comment.text}")
